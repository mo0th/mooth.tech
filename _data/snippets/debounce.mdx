---
title: debounce
short_description: Run a function at most once every N seconds, at the end of the N seconds.
category: TypeScript
tags:
  - typescript
created_at: '2021-03-28'
updated_at: '2021-03-28'
---

```typescript
const debounce = <T extends CallableFunction>(fn: T, delay: number): T => {
  let timer: ReturnType<typeof setTimeout>

  return ((...args: any[]) => {
    if (timer) clearTimeout(timer)
    setTimeout(() => {
      fn(...args)
    }, delay)
  }) as any
}
```
