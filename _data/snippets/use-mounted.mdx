Returns whether the current component is mounted. Useful for SSR stuff (e.g. with Next.js).

```typescript
import { useEffect, useState } from 'react'

const useMounted = (): boolean => {
  const [mounted, setMounted] = useState(false)

  useEffect(() => {
    setMounted(true)

    return () => {
      setMounted(false)
    }
  }, [])

  return mounted
}
```
